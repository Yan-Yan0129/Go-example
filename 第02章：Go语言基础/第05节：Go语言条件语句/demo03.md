```go
func main(){
    /*定义局部变量*/
    var a int = 100
    var b int = 100
    /*判断条件*/
    if a == 100{
        /*条件语句为true执行*/
        if b == 200{
            /*条件语句位true时执行*/
            fmt.printf("a 的值为 100 , b的值为200\n")
        }
    }
    fmt.printf("a值为 ： %d\n"  , a)
    fmt.printf("b值为 ： %d\n"  , b)
}
```
